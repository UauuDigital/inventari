function doGet(e) {
  const s = SpreadsheetApp
    .getActiveSpreadsheet().getSheets()[0];
  const p = e.parameter;
  s.appendRow([p.id, p.producte, p.proveidor,
               p.preu, p.categoria, p.codi]);
  return ContentService.createTextOutput('ok');
}
